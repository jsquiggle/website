  • 2

  • 3

Sample policy YAML
policy: cis_iosxe_baseline
version: 1.2.0
scope:
  vendors: [cisco, meraki, paloalto, fortinet]
  roles: [core, distribution, access, firewall]
checks:
  - id: no_telnet
    title: Disable Telnet
    when: vendor in [cisco, aruba]
    query: |
      show running-config | include line vty
    assert: not includes('transport input telnet')
    fix: |
      line vty 0 4
      transport input ssh
  - id: radius_timeout
    title: NPS Timeout >= 5s
    when: vendor == cisco
    query: |
      show run | sec radius-server
    assert: includes('timeout 5')
    fix: radius-server timeout 5
